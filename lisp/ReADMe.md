``` Lisp
(defun hasfeature (i l)
    (cond 
        ((null (symbol-plist i)) 'yes)
        ((equal (car (symbol-plist i)) (car l)) 
            (remprop i (car (symbol-plist i)))
            (hasfeature i (cddr l))) 
        (T
            'no
        )
    )
)
```
